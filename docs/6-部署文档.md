# 部署文档

## 1. 环境要求

- Docker 24.0+
- Docker Compose 2.20+
- 最低配置: 4核CPU, 8GB内存

## 2. 快速部署

### 2.1 开发环境

```bash
# 克隆项目
git clone <repository>
cd shortplay

# 启动基础设施 (MySQL, Redis, RabbitMQ, ES, MinIO, Qdrant)
docker compose -f docker/docker-compose.yml up -d

# 启动后端
cd backend
pip install -e .
uvicorn app.main:app --reload --port 8080

# 启动前端
cd frontend
npm install
npm run dev
```

### 2.2 生产环境

```bash
# 使用 Docker 部署全部服务
docker compose -f docker/docker-compose.prod.yml up -d
```

## 3. Docker Compose 配置

### 3.1 docker-compose.yml (开发环境)

```yaml
services:
  postgres:
    image: postgres:15
    container_name: shortplay-postgres
    restart: unless-stopped
    environment:
      POSTGRES_DB: shortplay
      POSTGRES_USER: shortplay
      POSTGRES_PASSWORD: ${DB_PASSWORD:-shortplay123}
      TZ: Asia/Shanghai
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U shortplay"]
      interval: 10s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7
    container_name: shortplay-redis
    restart: unless-stopped
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  mq:
    image: rabbitmq:4.2-management
    container_name: shortplay-mq
    restart: unless-stopped
    environment:
      RABBITMQ_DEFAULT_USER: shortplay
      RABBITMQ_DEFAULT_PASS: ${MQ_PASSWORD:-shortplay123}
      TZ: Asia/Shanghai
    ports:
      - "5672:5672"
      - "15672:15672"
    healthcheck:
      test: ["CMD", "rabbitmq-diagnostics", "-q", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  minio:
    image: minio/minio:latest
    container_name: shortplay-minio
    restart: unless-stopped
    environment:
      MINIO_ROOT_USER: ${MINIO_ACCESS_KEY:-minioadmin}
      MINIO_ROOT_PASSWORD: ${MINIO_SECRET_KEY:-minioadmin}
      TZ: Asia/Shanghai
    command: server /data --console-address ":9001"
    ports:
      - "9000:9000"
      - "9001:9001"
    volumes:
      - minio_data:/data
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9000/minio/health/live"]
      interval: 10s
      timeout: 5s
      retries: 5

  es:
    image: docker.1ms.run/library/elasticsearch:8.11.3
    container_name: shortplay-es
    restart: unless-stopped
    environment:
      discovery.type: single-node
      xpack.security.enabled: "false"
      ES_JAVA_OPTS: "-Xms512m -Xmx512m"
      TZ: Asia/Shanghai
    ports:
      - "9200:9200"
      - "9300:9300"
    volumes:
      - es_data:/usr/share/elasticsearch/data
    healthcheck:
      test: ["CMD-SHELL", "curl -fsS http://localhost:9200 >/dev/null || exit 1"]
      interval: 10s
      timeout: 5s
      retries: 10

  qdrant:
    image: qdrant/qdrant:latest
    container_name: shortplay-qdrant
    restart: unless-stopped
    environment:
      TZ: Asia/Shanghai
    ports:
      - "6333:6333"
      - "6334:6334"
    volumes:
      - qdrant_data:/qdrant/storage

volumes:
  postgres_data:
  redis_data:
  minio_data:
  es_data:
  qdrant_data:
```

### 3.2 docker-compose.prod.yml (生产环境)

```yaml
services:
  nginx:
    image: nginx:alpine
    container_name: shortplay-nginx
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/ssl:/etc/nginx/ssl:ro
    depends_on:
      - frontend
      - backend

  postgres:
    image: postgres:15
    # 生产环境使用外部托管数据库时注释此段
    deploy:
      resources:
        limits:
          memory: 2G
    # ...

  redis:
    image: redis:7-alpine
    # ...

  mq:
    image: rabbitmq:4.2-management
    # ...

  minio:
    image: minio/minio:latest
    # ...

  es:
    image: docker.1ms.run/library/elasticsearch:8.11.3
    # ...

  qdrant:
    image: qdrant/qdrant:latest
    # ...

  backend:
    build:
      context: ../backend
      dockerfile: Dockerfile
    container_name: shortplay-backend
    restart: unless-stopped
    environment:
      DATABASE_URL: postgresql://shortplay:${DB_PASSWORD}@postgres:5432/shortplay
      REDIS_URL: redis://redis:6379
      RABBITMQ_URL: amqp://shortplay:${MQ_PASSWORD}@mq:5672
      MINIO_ENDPOINT: minio:9000
      MINIO_ACCESS_KEY: ${MINIO_ACCESS_KEY}
      MINIO_SECRET_KEY: ${MINIO_SECRET_KEY}
      ELASTICSEARCH_URL: http://es:9200
      QDRANT_URL: qdrant:6333
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy
      mq:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  frontend:
    build:
      context: ../frontend
      dockerfile: Dockerfile
    container_name: shortplay-frontend
    restart: unless-stopped
    environment:
      VITE_API_BASE_URL: /api
    depends_on:
      - backend
```

## 5. Nginx 配置

```nginx
server {
    listen 80;
    server_name localhost;

    client_max_body_size 100M;

    location /api/ {
        proxy_pass http://backend:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    location /ws/ {
        proxy_pass http://backend:8080;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }

    location / {
        proxy_pass http://frontend:3000;
    }
}
```
